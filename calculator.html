        function calculateROI() {

            // Get form values

            const farmSize = parseFloat(document.getElementById('farmSize').value);

            const sensorCost = parseFloat(document.getElementById('sensorCost').value);

            const sensorCount = parseFloat(document.getElementById('sensorCount').value);

            const plantingArea = parseFloat(document.getElementById('plantingArea').value);

            const plantingCost = parseFloat(document.getElementById('plantingCost').value);

            const plantingMaintenance = parseFloat(document.getElementById('plantingMaintenance').value);

            const fertilizerSavings = parseFloat(document.getElementById('fertilizerSavings').value);

            const catchCropIncome = parseFloat(document.getElementById('catchCropIncome').value);

            const farmValuation = parseFloat(document.getElementById('farmValuation').value);

            const valuationIncreasePercent = parseFloat(document.getElementById('valuationIncreasePercent').value);

            const dataManagementCost = parseFloat(document.getElementById('dataManagementCost').value);

            const sensorMaintenance = parseFloat(document.getElementById('sensorMaintenance').value);

            const nitrateReductionPercent = parseFloat(document.getElementById('nitrateReductionPercent').value);


            // Validation

            if (farmSize <= 0 || sensorCount <= 0) {

                alert('Please enter valid positive numbers for farm size and sensor count.');

                return;

            }


            if (nitrateReductionPercent < 0 || nitrateReductionPercent > 100) {

                alert('Nitrate reduction percentage must be between 0-100%.');

                return;

            }


            // Calculations

            const sensorInstallCost = sensorCost * sensorCount;

            const plantingCapitalCost = plantingCost * plantingArea;

            const totalYear1Costs = sensorInstallCost + plantingCapitalCost;


            const annualSensorMaintenance = sensorMaintenance * sensorCount;

            const annualPlantingMaintenance = plantingMaintenance * plantingArea;

            const annualRecurringCosts = annualSensorMaintenance + annualPlantingMaintenance + dataManagementCost;


            const annualFertSavings = fertilizerSavings * farmSize;

            const annualCatchCropIncome = catchCropIncome * plantingArea;

            const annualValuationUplift = (farmValuation * (valuationIncreasePercent / 100)) / 10;


            // Generate results

            let cumulativeNet = 0;

            let tableHTML = '';


            for (let year = 1; year <= 5; year++) {

                const yearCosts = year === 1 ? totalYear1Costs : annualRecurringCosts;

                const yearBenefits = annualFertSavings + annualCatchCropIncome + annualValuationUplift;

                const net = yearBenefits - yearCosts;

                cumulativeNet += net;


                const costsClass = 'negative';

                const benefitsClass = 'positive';

                const netClass = net >= 0 ? 'positive' : 'negative';

                const cumulativeClass = cumulativeNet >= 0 ? 'positive' : 'negative';


                tableHTML += `

                    <tr>

                        <td>${year}</td>

                        <td class="${costsClass}">${formatCurrency(yearCosts)}</td>

                        <td class="${benefitsClass}">${formatCurrency(yearBenefits)}</td>

                        <td class="${netClass}">${formatCurrency(net)}</td>

                        <td class="${cumulativeClass}" style="font-weight: bold;">${formatCurrency(cumulativeNet)}</td>

                    </tr>

                `;

            }


            // Update display

            document.getElementById('nitrateReductionDisplay').textContent = nitrateReductionPercent + '%';

            document.getElementById('breakEvenDisplay').textContent = 

                cumulativeNet >= 0 ? 'Within 5 years' : 'More than 5 years - consider longer horizon';

            document.getElementById('resultsTableBody').innerHTML = tableHTML;

            document.getElementById('results').classList.remove('hidden');


            // Scroll to results on mobile

            document.getElementById('results').scrollIntoView({ behavior: 'smooth', block: 'start' });

        }


        function formatCurrency(amount) {

            return new Intl.NumberFormat('en-NZ', { 

                style: 'currency', 

                currency: 'NZD',

                minimumFractionDigits: 0,

                maximumFractionDigits: 0

            }).format(amount);

        }
